func jump(nums []int) int {
    n := len(nums)
    if n <= 1{
        return 0
    }
    steps , curEnd, maxReach := 0, 0, 0
    for i :=0 ; i< n ; i ++{
        maxReach = max(maxReach, i + nums[i])
        if i == curEnd{
            steps ++
            curEnd = maxReach
            if curEnd >= n-1{
                break
            }
        }
    }
    return steps
}
